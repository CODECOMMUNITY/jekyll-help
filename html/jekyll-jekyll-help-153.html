joewils Thursday, September 18, 2014

_layouts/post.html

{% capture path %}{{page.path}}{% endcapture %}
{% capture en_path %}{{ path | replace:'.md','.en.md' }}{% endcapture %}
{% capture cn_path %}{{ path | replace:'.md','.cn.md' }}{% endcapture %}
<div class="en">
    <h1>{{en_path}}</h1>
    {% include {{en_path}} %}
</div>
<div class="cn">
    <h1>{{cn_path}}</h1>
    {% include {{cn_path}} %}
</div>
<div class="default">
{{content}}
</div>

3 Files for every post

  • _posts/2014-09-18-test.md
  • _posts/2014-09-18-test.en.md
  • _posts/2014-09-18-test.cn.md
